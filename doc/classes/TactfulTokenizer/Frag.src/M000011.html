        def clean(s)
            @cleaned = String.new(s)
            tokenize(@cleaned)
            @cleaned.gsub!(/[.,\d]*\d/, '<NUM>')
            @cleaned.gsub!(/[^a-zA-Z0-9,.;:<>\-'\/$% ]/, '')
            @cleaned.gsub!('--', ' ')
            @cleaned = @cleaned.split
        end
